  <script>

    function updateTime() {

      const now = new Date();

      const timeStr = now.toLocaleTimeString('ko-KR', { hour12: false });

      const ms = now.getMilliseconds().toString().padStart(3, '0');

      document.getElementById('currentTime').innerHTML = timeStr + '.<span class="milliseconds">' + ms + '</span>';

    }

    setInterval(updateTime, 10);


    function getTimeInMS(date) {

      return date.getSeconds() * 1000 + date.getMilliseconds();

    }


    function submitApplication() {

      const button = document.getElementById('submitBtn');

      const spinner = document.getElementById('spinner');

      const complete = document.getElementById('completeScreen');


      button.disabled = true;

      spinner.style.display = 'block';


      const now = new Date();

      const myTime = getTimeInMS(now);


      const delay = Math.floor(Math.random() * 1000) + 500;


      setTimeout(() => {

        spinner.style.display = 'none';


        const base = 60 * 1000;

        const others = Array.from({length: 20000}, () => base + Math.floor(Math.random() * 3000) - 1000);

        others.push(myTime);

        others.sort((a, b) => a - b);

        const rank = others.indexOf(myTime) + 1;


        complete.style.display = 'block';

        complete.innerHTML = `✅ 신청 완료!<br>📊 당신의 예상 순위: <strong>${rank}</strong> / 20,001명`;

        button.disabled = false;

      }, delay);

    }

  </script>
